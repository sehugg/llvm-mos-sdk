88 /* Meaningful values for the distortion bits.
89 ** The first process is to divide the clock value by the frequency,
90 ** then mask the output using the polys in the order below;
91 ** finally, the result is divided by two.
92 */
93 #define AUDC_POLYS_5_17 0x00
94 #define AUDC_POLYS_5 0x20 /* Same as 0x60 */
95 #define AUDC_POLYS_5_4 0x40
96 #define AUDC_POLYS_17 0x80
97 #define AUDC_POLYS_NONE 0xA0 /* Same as 0xE0 */
98 #define AUDC_POLYS_4 0xC0
99 
100 /* When set, the volume value in AUDC1-4 bits 0-3 is sent directly to the speaker;
101 ** it is not modulated with the frequency specified in the AUDF1-4 registers.
102 ** (See "De Re Atari" Chapter 7: Sound)
103 */
104 #define AUDC_VOLUME_ONLY 0x10

139 #define SKCTL_KEYBOARD_DEBOUNCE 0x01 /* Enable keyboard debounce circuits */

142 /* Fast pot scan

146 #define SKCTL_FAST_POT_SCAN 0x04

148 /* POKEY two-tone mode

150 */

157 /* Bits 4, 5, and 6 of SKCTL set Serial Mode Control: */
