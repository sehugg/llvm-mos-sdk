55 struct __gtia_write {
56  unsigned char hposp0; /* 0x00: horizontal position of player 0 */
57  unsigned char hposp1; /* 0x01: horizontal position of player 1 */
58  unsigned char hposp2; /* 0x02: horizontal position of player 2 */
59  unsigned char hposp3; /* 0x03: horizontal position of player 3 */
60  unsigned char hposm0; /* 0x04: horizontal position of missile 0 */
61  unsigned char hposm1; /* 0x05: horizontal position of missile 1 */
62  unsigned char hposm2; /* 0x06: horizontal position of missile 2 */
63  unsigned char hposm3; /* 0x07: horizontal position of missile 3 */
64 
65  unsigned char sizep0; /* 0x08: size of player 0 */
66  unsigned char sizep1; /* 0x09: size of player 1 */
67  unsigned char sizep2; /* 0x0A: size of player 2 */
68  unsigned char sizep3; /* 0x0B: size of player 3 */
69  unsigned char sizem; /* 0x0C: size of missiles */
70 
71  unsigned char grafp0; /* 0x0D: graphics shape player 0 (used when ANTIC is not instructed to use DMA; see DMACTL) */
72  unsigned char grafp1; /* 0x0E: graphics shape player 1 */
73  unsigned char grafp2; /* 0x0F: graphics shape player 2 */
74  unsigned char grafp3; /* 0x10: graphics shape player 3 */
75  unsigned char grafm; /* 0x11: graphics shape missiles */
76 
77  unsigned char colpm0; /* 0x12: color player and missile 0 */
78  unsigned char colpm1; /* 0x13: color player and missile 1 */
79  unsigned char colpm2; /* 0x14: color player and missile 2 */
80  unsigned char colpm3; /* 0x15: color player and missile 3 */
81  unsigned char colpf0; /* 0x16: color playfield 0 */
82  unsigned char colpf1; /* 0x17: color playfield 1 */
83  unsigned char colpf2; /* 0x18: color playfield 2 */
84  unsigned char colpf3; /* 0x19: color playfield 3 */
85  unsigned char colbk; /* 0x1A: color background */
86 
87  unsigned char prior; /* 0x1B: priority selection */
88 
89  unsigned char vdelay;
90  /* 0x1C: vertical delay -- one-line resolution movement of
91  ** vertical position of an object when two line resolution display is enabled

93 
94  unsigned char gractl; /* 0x1D: stick/paddle latch, p/m control */
95 
96  unsigned char hitclr; /* 0x1E: clear p/m collision */
97  unsigned char consol; /* 0x1F: builtin speaker */

236 struct __gtia_read {
237  unsigned char m0pf; /* 0x00: missile 0 to playfield collision */

240  unsigned char m3pf; /* 0x03: missile 3 to playfield collision */
241  unsigned char p0pf; /* 0x04: player 0 to playfield collision */
242  unsigned char p1pf; /* 0x05: player 1 to playfield collision */
243  unsigned char p2pf; /* 0x06: player 2 to playfield collision */
244  unsigned char p3pf; /* 0x07: player 3 to playfield collision */
245  unsigned char m0pl; /* 0x08: missile 0 to player collision */
246  unsigned char m1pl; /* 0x09: missile 1 to player collision */
247  unsigned char m2pl; /* 0x0A: missile 2 to player collision */
248  unsigned char m3pl; /* 0x0B: missile 3 to player collision */
249  unsigned char p0pl; /* 0x0C: player 0 to player collision */
250  unsigned char p1pl; /* 0x0D: player 1 to player collision */
251  unsigned char p2pl; /* 0x0E: player 2 to player collision */
252  unsigned char p3pl; /* 0x0F: player 3 to player collision */
253 
254  unsigned char trig0; /* 0x10: joystick trigger 0 (0=pressed, 1=released) */
255  unsigned char trig1; /* 0x11: joystick trigger 1 */
256  unsigned char trig2; /* 0x12: joystick trigger 2 */
257  unsigned char trig3; /* 0x13: joystick trigger 3 */
258 
259  unsigned char pal; /* 0x14: pal/ntsc flag */
260 
261  unsigned char unused[10];
262 
263  unsigned char consol; /* 0x1F: console buttons */
264 };
