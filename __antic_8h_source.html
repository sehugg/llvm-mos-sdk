18 /* the generation of playfield graphics which is delivered as a datastream */
19 /* to the related CTIA/GTIA chip. The CTIA/GTIA provides the coloring of the */
20 /* playfield graphics, and is responsible for adding overlaid sprite */

57 /* Define a structure with the ANTIC coprocessor's register offsets */
58 /*****************************************************************************/
59 
60 struct __antic {
61  unsigned char dmactl; /* (W) direct memory access control */
62  unsigned char chactl; /* (W) character mode control */
63  unsigned char dlistl; /* display list pointer low-byte */
64  unsigned char dlisth; /* display list pointer high-byte */
65  unsigned char hscrol; /* (W) horizontal scroll enable */
66  unsigned char vscrol; /* (W) vertical scroll enable */
67  unsigned char unuse0; /* unused */
68  unsigned char pmbase; /* (W) msb of p/m base address (for when DMACTL has player and/or missile DMA enabled) */
69  unsigned char unuse1; /* unused */
70  unsigned char chbase; /* (W) msb of character set base address */
71  unsigned char wsync; /* (W) wait for horizontal synchronization */
72  unsigned char vcount; /* (R) vertical line counter */
73  unsigned char penh; /* (R) light pen horizontal position */
74  unsigned char penv; /* (R) light pen vertical position */
75  unsigned char nmien; /* (W) non-maskable interrupt enable */
76  union {
77  /* (W) ("NMIRES") nmi reset -- clears the interrupt request register;
78  ** resets all of the NMI status together
79  */
80  unsigned char nmires;
81 
82  /* (R) ("NMIST") nmi status -- holds cause for the NMI interrupt */
83  unsigned char nmist;
84  };
85 };

88 /*****************************************************************************/
89 /* DMACTL register options */
90 /*****************************************************************************/
91 
92 /* Initialized to 0x22: DMA fetch, normal playfield, no PMG DMA, double-line PMGs */
93 
94 /* Playfield modes: */
95 #define DMACTL_PLAYFIELD_NONE 0x00
96 #define DMACTL_PLAYFIELD_NARROW 0x01 /* e.g., 32 bytes per scanline with thick borders */
97 #define DMACTL_PLAYFIELD_NORMAL 0x02 /* e.g., 40 bytes per scanline with normal borders */
98 #define DMACTL_PLAYFIELD_WIDE 0x03 /* e.g., 48 bytes per scanline with no borders (overscan) */
99 
100 /* Other options: */
101 
102 /* If not set, GTIA's GRAFP0 thru GRAFP3, and/or GRAFM registers are used for
103 ** player & missile shapes, respectively. (Modify the registers during the horizontal blank
104 ** (Display List Interrupt), a la "racing the beam" on an Atari VCS/2600, )
105 ** if set, ANTIC's PMBASE will be used to fetch shapes from memory via DMA.
106 */
107 #define DMACTL_DMA_MISSILES 0x04
108 #define DMACTL_DMA_PLAYERS 0x08
109 
110 /* Unless set, PMGs (as fetched via DMA) will be double-scanline resolution */
111 #define DMACTL_PMG_SINGLELINE 0x10
112 
113 /* Unless set, ANTIC operation is disabled, since it cannot fetch
114 ** Display List instructions
115 */
116 #define DMACTL_DMA_FETCH 0x20
