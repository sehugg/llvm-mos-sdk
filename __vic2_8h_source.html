50 /* Define a structure with the vic register offsets. In cc65 mode, there
51 ** are aliases for the field accessible as arrays.
52 */
53 struct __vic2 {
54  union {
55  struct {
56  unsigned char spr0_x; /* Sprite 0, X coordinate */
57  unsigned char spr0_y; /* Sprite 0, Y coordinate */
58  unsigned char spr1_x; /* Sprite 1, X coordinate */
59  unsigned char spr1_y; /* Sprite 1, Y coordinate */
60  unsigned char spr2_x; /* Sprite 2, X coordinate */
61  unsigned char spr2_y; /* Sprite 2, Y coordinate */
62  unsigned char spr3_x; /* Sprite 3, X coordinate */
63  unsigned char spr3_y; /* Sprite 3, Y coordinate */
64  unsigned char spr4_x; /* Sprite 4, X coordinate */
65  unsigned char spr4_y; /* Sprite 4, Y coordinate */
66  unsigned char spr5_x; /* Sprite 5, X coordinate */
67  unsigned char spr5_y; /* Sprite 5, Y coordinate */
68  unsigned char spr6_x; /* Sprite 6, X coordinate */
69  unsigned char spr6_y; /* Sprite 6, Y coordinate */
70  unsigned char spr7_x; /* Sprite 7, X coordinate */
71  unsigned char spr7_y; /* Sprite 7, Y coordinate */
72  };
73  struct {
74  unsigned char x; /* X coordinate */
75  unsigned char y; /* Y coordinate */
76  } spr_pos[8];
77  };
78  unsigned char spr_hi_x; /* High bits of X coordinate */
79  unsigned char ctrl1; /* Control register 1 */
80  unsigned char rasterline; /* Current raster line */
81  union {
82  struct {
83  unsigned char strobe_x; /* Light pen, X position */
84  unsigned char strobe_y; /* Light pen, Y position */
85  };
86  struct {
87  unsigned char x; /* Light pen, X position */
88  unsigned char y; /* Light pen, Y position */
89  } strobe;
90  };
91  unsigned char spr_ena; /* Enable sprites */
92  unsigned char ctrl2; /* Control register 2 */
93  unsigned char spr_exp_y; /* Expand sprites in Y dir */
94  unsigned char addr; /* Address of chargen and video ram */
95  unsigned char irr; /* Interrupt request register */
96  unsigned char imr; /* Interrupt mask register */
97  unsigned char spr_bg_prio; /* Priority to background */
98  unsigned char spr_mcolor; /* Sprite multicolor bits */
99  unsigned char spr_exp_x; /* Expand sprites in X dir */
100  unsigned char spr_coll; /* Sprite/sprite collision reg */
101  unsigned char spr_bg_coll; /* Sprite/background collision reg */
102  unsigned char bordercolor; /* Border color */
103  union {
104  struct {
105  unsigned char bgcolor0; /* Background color 0 */
106  unsigned char bgcolor1; /* Background color 1 */
107  unsigned char bgcolor2; /* Background color 2 */
108  unsigned char bgcolor3; /* Background color 3 */
109  };
110  unsigned char bgcolor[4]; /* Background colors */
111  };
112  union {
113  struct {
114  unsigned char spr_mcolor0; /* Color 0 for multicolor sprites */
115  unsigned char spr_mcolor1; /* Color 1 for multicolor sprites */
116  };
117  /* spr_color is already used ... */
118  unsigned char spr_mcolors[2]; /* Color for multicolor sprites */
119  };
120  union {
121  struct {
122  unsigned char spr0_color; /* Color sprite 0 */
123  unsigned char spr1_color; /* Color sprite 1 */
124  unsigned char spr2_color; /* Color sprite 2 */
125  unsigned char spr3_color; /* Color sprite 3 */
126  unsigned char spr4_color; /* Color sprite 4 */
127  unsigned char spr5_color; /* Color sprite 5 */
128  unsigned char spr6_color; /* Color sprite 6 */
129  unsigned char spr7_color; /* Color sprite 7 */
130  };
131  unsigned char spr_color[8]; /* Colors for the sprites */
132  };
133 
134  /* The following ones are only valid in the C128: */
135  unsigned char x_kbd; /* Additional keyboard lines */
136  unsigned char clock; /* Clock switch bit */
137 };
